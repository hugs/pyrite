    def wait_for(element)
      case element
        when :page_load
          browser.wait_for_page_to_load :page
        when :ajax
          browser.wait_for(:wait_for => :ajax, :javascript_framework => Pyrite.js_framework)
        else
          browser.wait_for :element => "css=#{element}"
      end
    end
